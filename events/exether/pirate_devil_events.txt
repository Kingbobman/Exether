#A buyout
pirate_devil.2 = {
    type = country_event
    id = pirate_devil.2
    title = pirate_devil.2.t
    desc = pirate_devil.2.d
    flavor = pirate_devil.2.f
    popup = yes
	event_image = {
		video = "europenorthamerica_capitalists_meeting"
	}

    immediate = {
        random_country = {
            limit = {
                country_rank = rank_value:great_power
                relations:root > 60
            }
            save_scope_as = target_country
        }
    }

    option = {
        name = pirate_devil.2.a
        custom_tooltip = {
            text = pirate_devil.2.b.tt
        }
        scope:target_country = {
            trigger_event = {
                id = pirate_devil.3
                popup = yes
                days = 2
            }
        }
        add_modifier = {
            name = selling_ourselves
            months = 4
        }
    }

    option = {
        name = pirate_devil.2.b
        custom_tooltip = {
            text = pirate_devil.2.b.tt
        }
        add_modifier = {
            name = selling_ourselves
            days = 15
        }
    }
}
