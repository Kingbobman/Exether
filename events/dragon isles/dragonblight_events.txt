dragonblight.100 = { #kobold states receive this event the first time the Dragon Isles clicks the kobold settlement button and give them free land
type = country_event
placement = root
title = dragonblight.100.t
desc = dragonblight.100.d
flavor = dragonblight.100.f

duration = 3

event_image = {
	video = "unspecific_factory_closed" #change this
}

on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

icon = "gfx/interface/icons/event_icons/event_skull.dds"

trigger = { 

}

immediate = {
	every_scope_state = {
		limit = {
			NOT = {
				state_region = s:STATE_KLYPYRA
			}
		}
		remove_homeland = cu:illvasaari
		add_homeland = cu:kotskarvi
	}
	every_scope_state = {
		limit = {
			mainland_dragon_isles_state = yes
		}
		if={
			limit = {
				any_neighbouring_state = {	
						owner = {
							OR = {
								country_has_primary_culture = cu:kotskarvi
									country_has_primary_culture = cu:lomarash
								country_has_primary_culture = cu:drogori
								country_has_primary_culture = cu:mgzavri
								country_has_primary_culture = cu:garagoz
							}
						}
				}	
			}
			random_neighbouring_state = {
				limit = {
					owner = {
						OR = {
							country_has_primary_culture = cu:kotskarvi
								country_has_primary_culture = cu:lomarash
								country_has_primary_culture = cu:drogori
								country_has_primary_culture = cu:mgzavri
								country_has_primary_culture = cu:garagoz
						}
					}
				}
				owner = {
					save_scope_as = zamaris_secession_scope
				}
			}
			scope:outbreak_state = {
				dragonblight_killswitch_engage = yes
				set_state_owner = scope:zamaris_secession_scope
			}
		}
		else_if={
			limit = {
			#Relevant kobold tag
			exists = c:GUR
			}
			scope:outbreak_state = {
				dragonblight_killswitch_engage = yes
				set_state_owner = c:GUR
			}
		}
		else_if={
			limit = {
			#Relevant kobold tag
				exists = c:RKO
			}
			scope:outbreak_state = {
				dragonblight_killswitch_engage = yes
				set_state_owner = c:RKO
			}
		}	
		else = {
			scope:outbreak_state = {
				dragonblight_killswitch_engage = yes
			}
			create_country = {
				tag = GUR
				origin = this
				state = scope:outbreak_state.region_state:DIS
			}
		}
	}
	}
	option = { #gee thanks guys
	name = dragonblight.100.a
	default_option = yes

}
}
