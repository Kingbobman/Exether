magic_opinion_loc = {
	type = country

	text = {
		trigger = {
			var:arcana_magic_opinion >= 90
		}
		localization_key = concept_of_concept_magic_worshipped
	}
	text = {
		trigger = {
			var:arcana_magic_opinion < 90
			var:arcana_magic_opinion >= 60
		}
		localization_key = concept_of_concept_magic_approved
	}
	text = {
		trigger = {
			var:arcana_magic_opinion < 60
			var:arcana_magic_opinion >= 30
		}
		localization_key = concept_of_concept_magic_acceptable
	}
	text = {
		trigger = {
			var:arcana_magic_opinion < 30
			var:arcana_magic_opinion > -30
		}
		localization_key = concept_of_concept_magic_tolerable
	}
	text = {
		trigger = {
			var:arcana_magic_opinion <= -30
			var:arcana_magic_opinion > -60
		}
		localization_key = concept_of_concept_magic_unacceptable
	}
	text = {
		trigger = {
			var:arcana_magic_opinion <= -60
			var:arcana_magic_opinion > -90
		}
		localization_key = concept_of_concept_magic_hated
	}
	text = {
		trigger = {
			var:arcana_magic_opinion <= -90
		}
		localization_key = concept_of_concept_magic_abhorrent
	}
}
