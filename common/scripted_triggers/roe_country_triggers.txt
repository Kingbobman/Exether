discriminate_against_salt_waste_natives = {
	any_scope_pop = {
		culture = { has_discrimination_trait = saltwaste_survivor_culture_group }
		NOT = { is_pop_type = slaves }
		OR = {
			AND = {
				ROOT = { has_law = law_type:law_state_atheism }
				has_pop_religion = atheist
			}
			AND = {
				ROOT = { NOT = { has_law = law_type:law_state_atheism } }
				OR = {
					has_pop_religion = fathers_command
					has_pop_religion = mothers_embrace
					has_pop_religion = salt_crucible
				}
			}
		}
		pop_is_discriminated = yes
	}
}
