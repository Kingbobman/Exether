whiskers_of_change_VOZ_button = {
    name = "WoC_vozmead_bargain_button_name"
    desc = "WoC_vozmead_bargain_button_desc"

    visible = {
        exists = c:VOZ
        nor = {
            has_variable = vozmead_bargain_accepted
            has_variable = vozmead_bargain_denied
        }
    }

    possible = {
        nor = {
            has_variable = vozmead_bargain_offered
            has_variable = vozmead_bargain_accepted
            has_variable = vozmead_bargain_denied
        }
        government_legitimacy > 75
        ig:ig_petty_bourgeoisie = {
            is_in_government = yes
        }
        c:VOZ = {
            relations:root >= 20
        }
    }

    effect = {
        set_variable = vozmead_bargain_offered
        trigger_event = {
            id = whiskers_of_change.4
        }
    }

}
