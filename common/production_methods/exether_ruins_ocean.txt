	texture = "gfx/interface/icons/production_method_icons/potion_sap.dds"

	unlocking_technologies = {
		alchemy
	}

	building_modifiers = {
        workforce_scaled = {
            goods_input_potions_add = 10	# x50 = 500
            goods_input_hardwood_add = 3	# x40 = 120

            goods_output_treasure_add = 2
        }

        unscaled = {
            #building modifiers

    }
}
