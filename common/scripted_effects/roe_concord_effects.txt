effect_concord_application_approved = {
	remove_variable = concord_observer
	remove_variable = concord_supporting_list_size
	remove_variable = application_progress_concord
	clear_variable_list = concord_supporting
	clear_variable_list = concord_abstaining
	clear_variable_list = concord_rejecting
	remove_list_global_variable = {
		name = concord_application_list
		target = THIS
	}
	set_variable = concord_member
	set_country_type = recognized
}
effect_concord_application_rejected = {
	remove_variable = concord_supporting_list_size
	remove_variable = application_progress_concord
	clear_variable_list = concord_supporting
	clear_variable_list = concord_abstaining
	clear_variable_list = concord_rejecting
	remove_list_global_variable = {
		name = concord_application_list
		target = THIS
	}
	set_variable = {
		name = concord_application_rejected
		days = 365
	}
}

effect_reset_concord_lists = {
	set_global_variable = {
		name = pending_resolution_progress_concord
		value = 0
	}
	every_country = {
		if = {
			limit = {
				has_variable = is_supporting
			}
			remove_variable = is_supporting
		}
		if = {
			limit = {
				has_variable = is_supporting_concord
			}
			remove_variable = is_supporting_concord
		}
	}
	remove_global_variable = asking_concord_intervention_country
	remove_global_variable = vetoed_concord
	remove_global_variable = pending_resolution_concord
}

effect_ai_vote_yes_concord = {
	set_variable = is_supporting_concord
	remove_variable = is_opposing_concord
	set_variable = {
		name = concord_alreadyvoted
		days = 50
	}
}

effect_ai_vote_no_concord = {
	set_variable = is_opposing_concord
	remove_variable = is_supporting_concord
	set_variable = {
		name = concord_alreadyvoted
		days = 50
	}
}

effect_ai_vote_neutral_concord = {
	remove_variable = is_supporting_concord
	remove_variable = is_opposing_concord
	set_variable = {
		name = concord_alreadyvoted
		days = 50
	}
}

##### Rebuild UN list
rebuild_concord_memberlist = {
	clear_global_variable_list = concord_members_list
	every_country = {
		limit = {
			has_variable = concord_member
		}
		add_to_global_variable_list = {
			name = concord_members_list
			target = THIS
		}
	}
}
