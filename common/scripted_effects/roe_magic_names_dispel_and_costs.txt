get_cost = {
	if = {
		limit = {
			has_modifier = country_magic_tenet_upkeep_moderate
		}
		remove_modifier = country_magic_tenet_upkeep_moderate
		add_modifier = country_magic_tenet_upkeep_high
		
	}
	else_if = {
		limit = {
			has_modifier = country_magic_tenet_upkeep
		}
		remove_modifier = country_magic_tenet_upkeep
		add_modifier = country_magic_tenet_upkeep_moderate
		
	}
	else_if = {add_modifier = country_magic_tenet_upkeep}
}
