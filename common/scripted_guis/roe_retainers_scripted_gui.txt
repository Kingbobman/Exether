recruit_retainer = {
	saved_scopes = { recruited_character }
	effect = {
		if = {
			limit = {
				NOT = { has_variable = first_retainer }
			}
			set_variable = {
				name = first_retainer
				value = scope:recruited_character
			}
		}
		else_if = {
			limit = {
				NOT = { has_variable = second_retainer }
			}
			set_variable = {
				name = second_retainer
				value = scope:recruited_character
			}
		}
		else_if = {
			limit = {
				NOT = { has_variable = third_retainer }
			}
			set_variable = {
				name = third_retainer
				value = scope:recruited_character
			}
		}
		else_if = {
			limit = {
				NOT = { has_variable = fourth_retainer }
			}
			set_variable = {
				name = fourth_retainer
				value = scope:recruited_character
			}
		}
		else_if = {
			limit = {
				NOT = { has_variable = fifth_retainer }
			}
			set_variable = {
				name = fifth_retainer
				value = scope:recruited_character
			}
		}

		remove_list_variable = {
			name = available_retainers_list
			target = scope:recruited_character
		}

		create_character = {
			template = retainer_template
		}

	}
	ai_is_valid = {
		always = no
	}
	ai_chance = {
		base = 50
	}
	is_valid = {

		set_variable = {
			name = recruitment_balance
			value = renown
		}

		change_variable = {
			name = recruitment_balance
			subtract = 100
		}

		change_variable = {
			name = recruitment_balance
			subtract = scope:recruited_character.popularity
		}


		recruitment_balance >= 0
		NOT = { has_variable = fifth_retainer }
	}
}


show_placeholder_retainer = {

				scope:retainer_spot = 1
			}
			
		}
	}
}
