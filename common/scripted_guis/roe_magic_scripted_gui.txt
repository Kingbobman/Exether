# DIVINE #
cast_ritual_id_divine_button_effect = {
	saved_scopes = { ritual_id_divine }
	effect = {
		add_modifier = {
			name = country_ritual_cooldown
			months = 12
		}
		pick_ritual_id_divine = yes
		get_cost = yes
	}
	is_valid = {
		AND = {
			has_law = law_type:law_divine_miracles
			NOT = { 
				has_modifier = country_ritual_cooldown
			}
			scope:ritual_id_divine = {
				custom_tooltip = {
					text = has_reserves_tt
					root.magic_reserves_total >= ritual_cost
				}
			}
		}
       
    }
}
