raid = {
    is_shown = {
		root = {country_has_primary_culture = cu:south_sea_ravager}
	}

	possible = {
		root.var:raid_cooldown <= 0 
		NOT = { root.navy_size < 10}
	}
	
	when_taken = {
		set_variable = {
			name = raid_cooldown
            value = 3
		}

		trigger_event = {
			id = pirate_raid.0
			popup = yes
		}
	}
    
    on_monthly_pulse = {
        change_variable = {
            name = raid_cooldown
            subtract = 1
        }
    }
	
	ai_chance = {
		base = 1
	}
}
