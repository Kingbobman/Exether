locari_toldek_stop = {
    is_shown = {
        has_modifier = toldek_paseb
    }

    possible = {
        ig:ig_devout = {
            is_in_government = no
        }
        ig:ig_landowners = {
            is_in_government = no
        }
        ig:ig_mages = {
            is_in_government = no
        }
        government_legitimacy >= 70
        custom_tooltip = {
            text = bhimsen_died
            ruler = {
                NOT = {
                    has_variable = is_bhimsen
                }
            }
        }
    }

    when_taken = {
        remove_modifier = toldek_paseb
        add_modifier = {
            name = abolished_toldek_paseb
            months = normal_modifier_time
        }
	}
}

locari_examine_draconic_carvings = {
    is_shown = {
        any_primary_culture = {
            has_discrimination_trait = dragon_heritage
        }
    }

    possible = {
        owns_entire_state_region = STATE_ELENIKAR
        s:STATE_ELENIKAR = {
            any_scope_state = {
                any_scope_building = {
                    is_building_type = building_dragon_ruins
                    level = 8
			    }
            }
        }
    }

    when_taken = {
        trigger_event = {
            id = locari.1
            months = 3
        }
    }
}
